Create `packages/clerk/Dockerfile` — production-ready backend image:

2. Install UV for fast dependency resolution
3. Copy only necessary files (pyproject.toml, uv.lock, src/)

6. Use non-root user for security
7. Health check endpoint
8. CMD runs `uv run clerk web --host 0.0.0.0 --port 8000`

Structure:
```dockerfile
FROM python:3.13-slim

# Install UV
COPY --from=ghcr.io/astral-sh/uv:latest /uv /usr/local/bin/uv

# Create non-root user
RUN useradd --create-home --shell /bin/bash clerk

WORKDIR /app

# Copy dependency files
COPY pyproject.toml uv.lock ./
COPY src/ src/

# Install production dependencies
RUN uv sync --frozen --no-dev

# Switch to non-root user
USER clerk

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
  CMD python -c "import urllib.request; urllib.request.urlopen('http://localhost:8000/api/health')" || exit 1

EXPOSE 8000

CMD ["uv", "run", "clerk", "web", "--host", "0.0.0.0", "--port", "8000"]
```

  <name>Task 3: Create nginx reverse proxy configuration</name>
  <files>docker/nginx.conf</files>
  <action>
Create `docker/nginx.conf` — nginx configuration for production:

1. Serve static files for frontend routes
2. Proxy /api/* requests to backend service
3. Handle SPA routing (fallback to index.html for non-API routes)
4. Include security headers
5. Gzip compression for static assets

Configuration:
```nginx
server {
    listen 80;
    server_name localhost;

    # Gzip compression
    gzip on;
    gzip_types text/plain text/css application/json application/javascript;

    # Proxy to backend API
    location /api/ {
        proxy_pass http://backend:8000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # Static files
    location / {
        root /usr/share/nginx/html;
        try_files $uri $uri/ /index.html;
    }

    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
}
```

Per D-02: Frontend proxies API requests to backend container.

```yaml
version: '3.8'

services:
  db:
    image: postgres:16-alpine
    environment:
      POSTGRES_USER: ${POSTGRES_USER:-clerk}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD:-clerk}
      POSTGRES_DB: ${POSTGRES_DB:-clerk}
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_USER:-clerk}"]
      interval: 10s
      timeout: 5s
      retries: 5

  backend:
    build:
      context: ./packages/clerk
      dockerfile: Dockerfile
    env_file:
      - .env
    environment:
      DATABASE_URL: postgresql://${POSTGRES_USER:-clerk}:${POSTGRES_PASSWORD:-clerk}@db:5432/${POSTGRES_DB:-clerk}
    depends_on:
      db:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "python", "-c", "import urllib.request; urllib.request.urlopen('http://localhost:8000/api/health')"]
      interval: 30s
      timeout: 10s
      retries: 3

  frontend:
    build:
      context: ./apps/website
      dockerfile: Dockerfile
    ports:
      - "${FRONTEND_PORT:-80}:80"
    depends_on:
      - backend

volumes:
  postgres_data:
```

- Separate backend and frontend containers (per D-02)
- Health checks for db and backend
- .env file mounting (per D-04)
- Named volume for persistent database storage
